// Add a product
// to the 'on-order' list
public class AddProduct {
    int id;
    char[] name;
    String orderCode;

    public AddProduct(int id, char[] name, String orderCode) {
        this.id = id;
        this.name = name;
        this.orderCode = orderCode;
    }

    public int getId() {
        return id;
    }

    public void setId(int id) {
        this.id = id;
    }

    public char[] getName() {
        return name;
    }

    public void setName(char[] name) {
        this.name = name;
    }

    public String getOrderCode() {
        return orderCode;
    }

    public void setOrderCode(String orderCode) {
        this.orderCode = orderCode;
    }
}
